file(GLOB TS_FILES
    ${CMAKE_CURRENT_SOURCE_DIR}/*.ts
    )

add_custom_target(timzone-translations ALL
    DEPENDS ${TS_FILES}
    SOURCES ${TS_FILES}
    COMMAND ${LRELEASE} ${TS_FILES}
    COMMAND mkdir -p ${CMAKE_TIMEZONE_TRANSLATE_BUILD_PATH}
    COMMAND mv ${CMAKE_CURRENT_SOURCE_DIR}/*.qm ${CMAKE_TIMEZONE_TRANSLATE_BUILD_PATH}
)

install(DIRECTORY "${CMAKE_TIMEZONE_TRANSLATE_BUILD_PATH}/"
    DESTINATION ${CMAKE_TIMEZONE_TRANSLATE_INSTALL_PATH}
    FILES_MATCHING PATTERN "*.qm"
)
